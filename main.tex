\numberwithin{equation}{enumi}

\item show that \myvec{2\\3\\4},\myvec{-1\\-2\\1},and \myvec{5\\8\\7}  are collinear.

\solution\begin{enumerate}
    \item Let
    \begin{align}
        \vec{A}=\myvec{2\\3\\4}, \vec{B}=\myvec{-1\\-2\\1}, \vec{C}=\myvec{5\\8\\7}
    \end{align}\\
    Then,
    \begin{align}
        \vec{B}-\vec{A}=\myvec{-3\\-5\\-3}, \vec{C}-\vec{A}=\myvec{3\\5\\3}
    \end{align}
    and
    \begin{align}
     \vec{M}=\myvec{\vec{B}-\vec{A}&\vec{C}-\vec{A}}^T\end{align}
     \begin{align}
         =\myvec{-3 & -5 & -3\\3 & 5 & 3} \xleftrightarrow{R_2 \leftarrow R_1+R_2}
     \myvec{-3 & -5 & -3\\0 & 0 & 0}
     \end{align}
         \[\Longrightarrow \emph{rank}(\vec{M})=1 \Longleftrightarrow R_2 = \vec{0}\]
\end{enumerate}
